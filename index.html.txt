    <h2>LOG DATA PRODUCTION 1 - M RIVALDI EKA SATRIA</h2>
    <div id="status">Status: Menghubungkan...</div>
    <div id="output"></div>

    <script>
        const apiURL = "https://script.google.com/macros/s/AKfycbyOfYnDgbRJhJa9FfoAuuNkHS6zGuH87JsaONhqY34NMQhaUr6pifNeUYxlWcWH6vbw6g/exec";

        async function getLog() {
            try {
                const res = await fetch(apiURL);
                const data = await res.json();
                document.getElementById('status').innerHTML = "Status: Terhubung (Real-time)";
                
                let html = "";
                
                // Ambil 5 data terakhir dari tab Average Temp
                const tempLogs = data["Average Temp"].slice(-5).reverse();
                html += "<h3>--- 5 LOG SUHU TERAKHIR ---</h3>";
                tempLogs.forEach(log => {
                    html += `<div class="data-box">TANGGAL: ${log.Tanggal} | <span class="val">SUHU: ${log.Suhu}°C</span></div>`;
                });

                // Ambil 5 data terakhir dari tab HM
                const hmLogs = data["HM & Ampere Average"].slice(-5).reverse();
                html += "<h3>--- 5 LOG HM & AMPERE TERAKHIR ---</h3>";
                hmLogs.forEach(log => {
                    html += `<div class="data-box">TANGGAL: ${log.Tanggal} | HM: ${log.HM} | AMP: ${log.Ampere}</div>`;
                });

                document.getElementById('output').innerHTML = html;
            } catch (e) {
                document.getElementById('status').innerHTML = "Status: Gagal ambil data.";
            }
        }
        getLog();
        setInterval(getLog, 60000); // Auto update setiap 1 menit
    </script>

<script>
// URL API yang baru saja Anda buat
const apiURL = "https://script.google.com/macros/s/AKfycbyOfYnDgbRJhJa9FfoAuuNkHS6zGuH87JsaONhqY34NMQhaUr6pifNeUYxlWcWH6vbw6g/exec";

async function updateDashboard() {
try {
const response = await fetch(apiURL);
const allData = await response.json();
document.getElementById('status-loading').style.display = 'none';

// 1. DATA SUHU (Dari Tab "Average Temp")
const dataSuhu = allData["Average Temp"].slice(-7); // Ambil 7 data terakhir
const labelsSuhu = dataSuhu.map(item => item["Tanggal"] || "N/A");
const nilaiSuhu = dataSuhu.map(item => item["Suhu"] || 0);
renderChart('chartSuhu', labelsSuhu, nilaiSuhu, 'Suhu Rata-rata', '#e74c3c');

// 2. DATA HM & AMPERE (Dari Tab "HM & Ampere Average")
const dataHMAmpere = allData["HM & Ampere Average"].slice(-7);
const labelsHMAmpere = dataHMAmpere.map(item => item["Tanggal"] || "N/A");

// Kolom HM (Sesuaikan nama kolom di sheet Anda)
const nilaiHM = dataHMAmpere.map(item => item["HM"] || 0);
renderChart('chartHM', labelsHMAmpere, nilaiHM, 'Total HM', '#3498db');

// Kolom Ampere
const nilaiAmpere = dataHMAmpere.map(item => item["Ampere"] || 0);
renderChart('chartAmpere', labelsHMAmpere, nilaiAmpere, 'Avg Ampere', '#2ecc71');

} catch (error) {
document.getElementById('status-loading').innerHTML = "❌ Gagal memuat data. Cek kembali koneksi API.";
console.error(error);
}
}

function renderChart(id, labels, data, title, color) {
new Chart(document.getElementById(id), {
type: 'line',
data: {
labels: labels,
datasets: [{
label: title,
data: data,
borderColor: color,
backgroundColor: color + '22',
fill: true,
tension: 0.4
}]
}
});
}

updateDashboard();

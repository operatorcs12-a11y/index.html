<script>
    // Link Apps Script Anda
    const apiURL = "https://script.google.com/macros/s/AKfycbyOfYnDgbRJhJa9FfoAuuNkHS6zGuH87JsaONhqY34NMQhaUr6pifNeUYxlWcWH6vbw6g/exec";

    async function getData() {
        try {
            const response = await fetch(apiURL);
            const data = await response.json();
            document.getElementById('loader').style.display = 'none';

            // 1. Ambil dari Tab "Average Temp" (SESUAI GAMBAR)
            const dsTemp = data["Average Temp"].slice(-7); 
            const labelTemp = dsTemp.map(d => d["Tanggal"]);
            const valTemp = dsTemp.map(d => d["Suhu"] || 0);
            renderChart('chartSuhu', labelTemp, valTemp, 'Suhu', '#e74c3c');

            // 2. Ambil dari Tab "HM & Ampere Average" (SESUAI GAMBAR)
            const dsHM = data["HM & Ampere Average"].slice(-7);
            const labelHM = dsHM.map(d => d["Tanggal"]);
            const valHM = dsHM.map(d => d["HM"] || 0);
            renderChart('chartHM', labelHM, valHM, 'Running Hours', '#3498db');

        } catch (error) {
            document.getElementById('loader').innerHTML = "Gagal memuat data. Cek Apps Script!";
            console.error(error);
        }
    }

    function renderChart(id, labels, values, title, color) {
        new Chart(document.getElementById(id), {
            type: 'line',
            data: {
                labels: labels,
                datasets: [{
                    label: title,
                    data: values,
                    borderColor: color,
                    backgroundColor: color + '22',
                    fill: true,
                    tension: 0.3
                }]
            }
        });
    }

    getData();
</script>
